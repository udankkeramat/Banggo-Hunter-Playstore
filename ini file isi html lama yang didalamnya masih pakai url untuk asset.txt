        function handlePhotoUpload(event) {
            playSound('ui');
            const file = event.target.files[0];
            if (!file) return;
            
            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    canvas.width = 90; canvas.height = 90;
                    const ctx = canvas.getContext('2d');
                    const size = Math.min(img.width, img.height);
                    const sx = (img.width - size) / 2;
                    const sy = (img.height - size) / 2;
                    ctx.drawImage(img, sx, sy, size, size, 0, 0, 90, 90);
                    
                    tempPhotoBase64 = canvas.toDataURL('image/jpeg', 0.8);
                    document.getElementById('photoConfirmPreview').src = tempPhotoBase64;
                    openModal('photoConfirmLayer');
                };
                img.src = e.target.result;
            };
            reader.readAsDataURL(file);
            event.target.value = '';
        }

        async function saveScoreAPI(id, name, score, accuracy, photo) {
            let localData = JSON.parse(localStorage.getItem('banggo_lb_rank')) || []; 
            let existing = localData.find(x => x.id === id);
            if(existing) { if(score > existing.score) { existing.score = score; existing.name = name; existing.accuracy = accuracy; existing.photo = photo; } } 
            else { localData.push({id: id, name: name, score: score, accuracy: accuracy, photo: photo}); }
            localData.sort((a,b) => b.score - a.score); localStorage.setItem('banggo_lb_rank', JSON.stringify(localData.slice(0, 31)));
            
            try { 
                if (SCRIPT_URL.includes('script.google.com')) { 
                    let fd = new FormData(); 
                    fd.append('id', id); fd.append('name', name); fd.append('score', score); fd.append('accuracy', accuracy); fd.append('photo', photo);
                    await fetch(SCRIPT_URL, { method: 'POST', body: fd, mode: 'no-cors' }); 
                } 
            } catch(e) { throw e; }
        }

        class Bird {
            constructor() { this.reset(); }
            reset() {
                this.active = true; this.state = 'flying'; 
                
                let multiplier50 = Math.floor(score / 50);
                let multiplier100 = Math.floor(score / 100);
                
                let speedBoost = Math.pow(1.1, multiplier50); 
                let scaleShrink = Math.pow(0.9, multiplier100); 
                let goldChance = Math.max(0.01, 0.1 - (multiplier100 * 0.02));

                this.isGold = Math.random() < goldChance; 
                if (activeSkill === 'elena') this.isGold = true;
                this.type = this.isGold ? 'gold' : 'normal';
                
                let rawScale = 0.25 + Math.random() * 0.35;
                this.scale = rawScale * scaleShrink;
                this.baseSize = 400 * this.scale; 
                this.hitboxRadius = this.baseSize * 0.35;
                
                const spawnRight = Math.random() > 0.5; const spawnOffset = canvas.width / 2 + 400;
                this.x = cameraOffset.x + (spawnRight ? spawnOffset : -spawnOffset); 
                this.y = cameraOffset.y - (canvas.height * 0.2) + (Math.random() * canvas.height * 0.3);
                this.direction = spawnRight ? -1 : 1; 
                
                let baseSpd = (Math.random() * (BASE_SPAWN.speedMax - BASE_SPAWN.speedMin) + BASE_SPAWN.speedMin) * (rawScale + 0.5) * speedBoost;
                if (gameplayType === 'frenzy') baseSpd *= 1.5;
                
                this.speed = baseSpd; if(this.isGold) this.speed *= 1.2;
                this.bobOffset = Math.random() * Math.PI * 2; this.bobSpeed = 0.002 + Math.random() * 0.004;
                this.animFly = new SpriteAnim(this.isGold ? images.goldFly : images.banggoFly, 8, ANIM_SPEED_0_1S, true); this.animDown = new SpriteAnim(this.isGold ? images.goldDown : images.banggoDown, 4, ANIM_SPEED_0_1S, true);
            }
            update() {
                if (!this.active) return;
                if (activeSkill === 'silas' && this.state === 'flying') return; 
                if (this.state === 'flying') { 
                    let currentSpeed = this.speed; 
                    if (activeSkill === 'eab') currentSpeed *= 0.5; 
                    if (activeSkill === 'elena') currentSpeed *= 0.7; 
                    
                    this.x += currentSpeed * this.direction; 
                    this.y += Math.sin(Date.now() * this.bobSpeed + this.bobOffset) * 2; 
                    this.animFly.update(); 
                    if (Math.abs(this.x - cameraOffset.x) > canvas.width * 2) this.active = false; 
                }
                else if (this.state === 'dying') { 
                    this.y += 12; this.x += this.direction * 2; 
                    this.animDown.update(); 
                    if (this.y - cameraOffset.y > canvas.height + 600) this.active = false; 
                }
            }
            render(ctx, camX, camY) {
                if (!this.active) return; const drawX = this.x - camX; const drawY = this.y - camY;
                if (drawX < -300 || drawX > canvas.width + 300 || drawY < -600 || drawY > canvas.height + 600) return;
                const shouldFlip = this.direction === -1;
                
                let drawSize = this.baseSize;
                if (activeSkill === 'elena') drawSize *= 2; 
                
                if (this.state === 'flying') { 
                    const ar = this.animFly.frameWidth / this.animFly.frameHeight; 
                    this.animFly.draw(ctx, drawX, drawY, drawSize * ar, drawSize, shouldFlip); 
                } 
                else { 
                    const ar = this.animDown.frameWidth / this.animDown.frameHeight; 
                    const drawW = drawSize * 0.8; 
                    this.animDown.draw(ctx, drawX, drawY, drawW, drawW / ar, shouldFlip); 
                }
            }
            checkHit(camX, camY, chOffsetY) { 
                if (this.state !== 'flying') return false; 
                let currentHitbox = this.hitboxRadius; 
                if (activeSkill === 'banggo') currentHitbox *= 2; 
                if (activeSkill === 'elena') currentHitbox *= 2; 
                if (activeSkill === 'jack') currentHitbox *= 4; 
                return Math.hypot((this.x - camX) - 0, (this.y - camY) - chOffsetY) < currentHitbox; 
            }
            die() { this.state = 'dying'; playSound('hit'); effects.push(new ShotEffect(this.x, this.y)); }
        }

        function handleTouchStart(e) { 
            if (gameState !== 'PLAY') return; if (e.target !== canvas) return; e.preventDefault(); 
            const charObj = charData.find(c => c.id === equippedCharId);
            let isRemiSkillDisabled = (equippedCharId === 'remi' && gameplayType === 'sniper' && (sniperAmmo >= 5 || remiSniperSkillCount >= 5));
            
            for (let i = 0; i < e.changedTouches.length; i++) { 
                const t = e.changedTouches[i]; 
                if (t.clientX > shotBtnRect.x && t.clientX < shotBtnRect.x + shotBtnRect.w && t.clientY > shotBtnRect.y && t.clientY < shotBtnRect.y + shotBtnRect.h) { isShotBtnPressed = true; shoot(); continue; } 
                if (t.clientX > skillBtnRect.x && t.clientX < skillBtnRect.x + skillBtnRect.w && t.clientY > skillBtnRect.y && t.clientY < skillBtnRect.y + skillBtnRect.h && (!charObj || !charObj.isPassive) && !isRemiSkillDisabled) { isSkillBtnPressed = true; activateSkill(); continue; } 
                if (t.clientX < canvas.width / 2) { analogInput.activeTouch = true; analogInput.id = t.identifier; analogInput.originX = t.clientX; analogInput.originY = t.clientY; analogInput.currX = t.clientX; analogInput.currY = t.clientY; } 
            } 
        }
        function handleTouchMove(e) { if (gameState !== 'PLAY') return; e.preventDefault(); for (let i = 0; i < e.changedTouches.length; i++) { const t = e.changedTouches[i]; if (analogInput.activeTouch && t.identifier === analogInput.id) { analogInput.currX = t.clientX; analogInput.currY = t.clientY; let dx = analogInput.currX - analogInput.originX; let dy = analogInput.currY - analogInput.originY; const maxRadius = 60; const dist = Math.hypot(dx, dy); if (dist > maxRadius) { const angle = Math.atan2(dy, dx); dx = Math.cos(angle) * maxRadius; dy = Math.sin(angle) * maxRadius; } analogInput.x = dx / maxRadius; analogInput.y = dy / maxRadius; } } }
        function handleTouchEnd(e) { for (let i = 0; i < e.changedTouches.length; i++) { const t = e.changedTouches[i]; if (analogInput.activeTouch && t.identifier === analogInput.id) { analogInput.activeTouch = false; analogInput.x = 0; analogInput.y = 0; } isShotBtnPressed = false; isSkillBtnPressed = false; } }

        function shoot() {
            const now = Date.now(); 
            let currentCooldown = SHOT_COOLDOWN_MS; 
            if (activeSkill !== 'jack' && now < shotCooldownTime) return;

            shotsFired++; screenShake = 10; 
            if (activeSkill === 'jack') shotCooldownTime = 0; else shotCooldownTime = now + currentCooldown; 
            playShotSound();
            crosshairScale = 1.4; crosshairRotation = (Math.random() - 0.5) * 0.5; crosshairOffset.y -= 25; if(crosshairOffset.y < -canvas.height*0.4) crosshairOffset.y = -canvas.height*0.4;

            let hit = false; let hitBird = null;

            if (activeSkill === 'ikra') {
                let onScreenBirds = birds.filter(b => {
                    if (!b.active || b.state !== 'flying') return false;
                    const drawX = b.x - cameraOffset.x;
                    return (drawX > -100 && drawX < canvas.width + 100);
                });
                if (onScreenBirds.length > 0) {
                    hit = true; hitBird = onScreenBirds[Math.floor(Math.random() * onScreenBirds.length)];
                    crosshairOffset.y = (hitBird.y - cameraOffset.y) - canvas.height/2; hitBird.die();
                }
            } 
            else {
                for (let i = birds.length - 1; i >= 0; i--) { 
                    const b = birds[i]; 
                    if (b.checkHit(cameraOffset.x, cameraOffset.y, crosshairOffset.y)) { hit = true; hitBird = b; b.die(); if (activeSkill !== 'jack') break; } 
                }
            }
            
            if (hit) {
                hits++; if (now - lastHitTime > COMBO_TIMEOUT_MS) { comboCount = 0; fadeOutComboBGM(); } lastHitTime = now; comboCount++;
                let pointsAdded = 0; let popText = "";
                
                if (hitBird.isGold) { 
                    pointsAdded = 10; score += pointsAdded; popText = "GOLD BANGGO DOWN!"; playSound('gold_banggo_down'); 
                    if (Math.random() < 0.05) { banggoFeathers++; saveRPGData(); showTopNotif(images.iconFeather.src, '1 Golden Feather\nDrop!'); }
                } else { 
                    pointsAdded = (comboCount > 5) ? 6 : comboCount; score += pointsAdded; 
                    if (comboCount === 2) { popText = "DOUBLE SHOT"; playSound('doubleshot'); } else if (comboCount === 3) { popText = "TRIPPLE SHOT"; playSound('trippleshot'); } else if (comboCount === 4) { popText = "SKY REAPER"; playSound('sky_reaper'); } else if (comboCount === 5) { popText = "FEATHER FRENZY"; playSound('feather_frenzy'); } else if (comboCount > 5) { popText = "BANGGO SLAYER!"; playSound('banggo_slayer'); } 
                }

                if (comboCount >= 3) { const cv = document.getElementById('comboVideo'); if(cv && cv.paused) { cv.playPromise = cv.play(); if(cv.playPromise !== undefined) cv.playPromise.catch(()=>{}); } playComboBGM(comboCount >= 5 ? 1.0 : 0.5); }
                
                if (activeSkill === 'risa') { score += pointsAdded; pointsAdded *= 2; popText += " (x2)"; }
                if (activeSkill === 'elena') { score += pointsAdded; pointsAdded *= 2; popText += " (x2)"; }
                if (activeSkill === 'edy') { banggoCoin += 2; saveRPGData(); popText += " (+2 🪙)"; }
                
                scorePopups.push(new ScorePopup(`+${pointsAdded}`, hitBird.x, hitBird.y));
                updateHUD(); if(popText) showFloatingText(popText); crosshairScale = 1.6;

                if (gameplayType === 'frenzy' && comboCount > 0 && comboCount % 5 === 0) { timeLeft += 2; showFrenzyTimeAdd(); updateHUD(); }

            } else { 
                let immuneToMiss = false;
                if (activeSkill === 'jack') immuneToMiss = true;
                if (activeSkill === 'silas') immuneToMiss = true;
                if (activeSkill === 'ikra') immuneToMiss = true;
                if (equippedCharId === 'lyra' && Date.now() > lyraCooldownTimer) { immuneToMiss = true; lyraCooldownTimer = Date.now() + 15000; showFloatingText("Auto-Correction Shield!"); }
                if (equippedCharId === 'remi' && Date.now() < remiSuddenImmunityEnd) immuneToMiss = true;

                if (immuneToMiss) return; 

                if (gameplayType === 'sniper') { sniperAmmo--; updateAmmoUI(); if (sniperAmmo <= 0) endGame(); } 
                else if (gameplayType === 'sudden') { endGame(); }

                comboCount = 0; fadeOutComboBGM(); 
            }
        }

                ctx.restore();

                const now = Date.now();
                if (now < skillCooldownEndTime && !isRemiSkillDisabled) { const stats = getSkillStats(equippedCharId); const percent = (skillCooldownEndTime - now) / stats.cd; ctx.beginPath(); ctx.moveTo(sDrawX + sDrawW/2, sDrawY + sDrawH/2); ctx.arc(sDrawX + sDrawW/2, sDrawY + sDrawH/2, sDrawW/2, -Math.PI/2, -Math.PI/2 + (Math.PI * 2 * percent), false); ctx.closePath(); ctx.fillStyle = 'rgba(0, 0, 0, 0.7)'; ctx.fill(); }
            }

            if (analogInput.activeTouch) { ctx.beginPath(); ctx.arc(analogInput.originX, analogInput.originY, 60, 0, Math.PI*2); ctx.fillStyle = 'rgba(255, 255, 255, 0.1)'; ctx.lineWidth = 2; ctx.strokeStyle = 'rgba(255, 255, 255, 0.3)'; ctx.stroke(); ctx.fill(); ctx.beginPath(); ctx.arc(analogInput.currX, analogInput.currY, 25, 0, Math.PI*2); ctx.fillStyle = 'rgba(156, 163, 175, 0.8)'; ctx.fill(); } 
            else if ('ontouchstart' in window) { ctx.beginPath(); ctx.arc(analogInput.originX, analogInput.originY, 40, 0, Math.PI*2); ctx.fillStyle = 'rgba(255, 255, 255, 0.1)'; ctx.fill(); ctx.strokeStyle = 'rgba(255, 255, 255, 0.1)'; ctx.stroke(); }
        }

        function gameLoop(timestamp) {
            const dt = timestamp - lastTime; lastTime = timestamp; ctx.clearRect(0, 0, canvas.width, canvas.height);
            if (images.bg) drawParallaxBG(ctx);
            const viewCenterX = cameraOffset.x - canvas.width/2; const viewCenterY = cameraOffset.y - canvas.height/2;
            
            if (activeSkill && Date.now() > skillEndTime) activeSkill = null;

            if (gameState === 'PLAY') {
                const sensitivity = 8; 
                if (analogInput.activeTouch) { cameraOffset.x += analogInput.x * sensitivity; cameraOffset.y += analogInput.y * sensitivity; }
                if (cameraOffset.y < -CAMERA_Y_LIMIT) cameraOffset.y = -CAMERA_Y_LIMIT; if (cameraOffset.y > CAMERA_Y_LIMIT) cameraOffset.y = CAMERA_Y_LIMIT;
                crosshairOffset.y *= 0.9;
                
                spawnManager(); birds.forEach(b => b.update()); bgBirds.forEach(bb => bb.update()); clouds.forEach(c => c.update()); effects.forEach(e => e.update()); scorePopups.forEach(sp => sp.update());
                
                for(let i=effects.length-1; i>=0; i--) if(!effects[i].active) effects.splice(i,1);
                for(let i=scorePopups.length-1; i>=0; i--) if(!scorePopups[i].active) scorePopups.splice(i,1);
                for(let i=bgBirds.length-1; i>=0; i--) if(!bgBirds[i].active) bgBirds.splice(i,1);
                
                if(screenShake > 0) screenShake *= 0.9; if(screenShake < 0.5) screenShake = 0;
                crosshairScale += (1.0 - crosshairScale) * 0.2; crosshairRotation *= 0.8;
                if(comboCount > 0 && Date.now() - lastHitTime > COMBO_TIMEOUT_MS) { comboCount = 0; fadeOutComboBGM(); }
            }
            if(gameState === 'PLAY' || gameState === 'MENU' || gameState === 'COUNTDOWN' || gameState === 'END' ) { clouds.forEach(c => c.render(ctx, viewCenterX, viewCenterY)); }
            if(gameState === 'PLAY' || gameState === 'MENU' || gameState === 'COUNTDOWN' || gameState === 'END' ) {
                ctx.save(); let shakeX = (Math.random() - 0.5) * screenShake; let shakeY = (Math.random() - 0.5) * screenShake; ctx.translate(shakeX, shakeY);
                bgBirds.forEach(bb => bb.render(ctx, viewCenterX, viewCenterY));
                const sortedBirds = [...birds].sort((a, b) => a.scale - b.scale); sortedBirds.forEach(b => b.render(ctx, viewCenterX, viewCenterY));
                effects.forEach(e => e.render(ctx, viewCenterX, viewCenterY)); scorePopups.forEach(sp => sp.render(ctx, viewCenterX, viewCenterY));
                ctx.restore();
            }
            if(gameState === 'MENU' || gameState === 'LOADING' || gameState === 'WAITING_START' || gameState === 'COUNTDOWN' || gameState === 'END') { ctx.fillStyle = 'rgba(0,0,0,0.7)'; ctx.fillRect(0,0,canvas.width, canvas.height); }
            if (gameState === 'PLAY' || gameState === 'PAUSE') {
                if (images.scope || images.crosshair) { if (crosshairEntryParams.progress < 0.99) crosshairEntryParams.progress += (1 - crosshairEntryParams.progress) * 0.1; else crosshairEntryParams.progress = 1; }
                const entryScaleFactor = 1 + (9 * (1 - crosshairEntryParams.progress)); const entryAlpha = crosshairEntryParams.progress;

                if (images.scope) {
                    ctx.save(); const chX = canvas.width/2; const chY = canvas.height/2 + crosshairOffset.y; ctx.translate(chX, chY); ctx.scale(crosshairScale * entryScaleFactor, crosshairScale * entryScaleFactor); ctx.rotate(crosshairRotation); ctx.globalAlpha = entryAlpha;
                    const sx = 0; const sy = 68; const sW = images.scope.width; const sH = images.scope.height - 136; const aspect = sW / sH; let drawW = canvas.width; let drawH = canvas.width / aspect; if (drawH < canvas.height) { drawH = canvas.height; drawW = canvas.height * aspect; } drawW *= 1.2; drawH *= 1.2; 
                    ctx.drawImage(images.scope, sx, sy, sW, sH, -drawW/2, -drawH/2, drawW, drawH); 
                    
                    if (activeSkill && skillEndTime > Date.now()) {
                        const charObj = charData.find(c => c.id === activeSkill);
                        if (charObj && charObj.duration > 0 && images.skillGauge) {
                            const stats = getSkillStats(activeSkill);
                            const percent = Math.max(0, (skillEndTime - Date.now()) / stats.dur);
                            const gW = images.skillGauge.width; const gH = images.skillGauge.height;
                            const dGaugeW = gW * (drawW / sW); const dGaugeH = gH * (drawH / sH);
                            const dX = -drawW/2 + (1253 / sW) * drawW;
                            const dY = -drawH/2 + ((411 - 68) / sH) * drawH; 
                            
                            const srcY = gH * (1 - percent);
                            const srcH = gH * percent;
                            const destY = dY + dGaugeH * (1 - percent);
                            const destH = dGaugeH * percent;
                            
                            if (srcH > 0) {
                                ctx.drawImage(images.skillGauge, 0, srcY, gW, srcH, dX, destY, dGaugeW, destH);
                            }
                        }
                    }

                    ctx.restore();
                }

                const cv = document.getElementById('comboVideo'); if (comboCount >= 3 && !cv.paused && cv.readyState >= 2) { ctx.save(); ctx.globalAlpha = 0.5; ctx.globalCompositeOperation = 'screen'; ctx.drawImage(cv, 0, 0, canvas.width, canvas.height); ctx.restore(); }
                if (images.crosshair) {
                    let chSize = 100 * crosshairScale * entryScaleFactor; ctx.save(); const chX = canvas.width/2; const chY = canvas.height/2 + crosshairOffset.y; ctx.translate(chX, chY); ctx.rotate(crosshairRotation); ctx.globalAlpha = entryAlpha;
                    if (activeSkill === 'banggo') { chSize *= 1.5; ctx.shadowColor = "#ff0000"; ctx.shadowBlur = 20; }
                    ctx.drawImage(images.crosshair, -chSize/2, -chSize/2, chSize, chSize); ctx.restore();
                } 
                drawControls(ctx);
            }
            requestAnimationFrame(gameLoop);
        }

        resize(); initGame(); requestAnimationFrame(gameLoop);

    </script>
